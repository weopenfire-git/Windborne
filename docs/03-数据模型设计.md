## 2. 表结构详细设计

### 2.1 users · 用户资料

```sql
CREATE TABLE users (
  id          UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  email       TEXT NOT NULL,
  username    TEXT UNIQUE NOT NULL,
  nickname    TEXT NOT NULL DEFAULT '',
  avatar_url  TEXT,
  bio         TEXT DEFAULT '',
  role        TEXT NOT NULL DEFAULT 'member' CHECK (role IN ('member', 'verified_aviator', 'admin')),
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- 自动从 auth.users 同步 email
COMMENT ON TABLE users IS '用户资料表，id 与 Supabase auth.users 关联';
```

| 字段 | 类型 | 说明 |
|------|------|------|
| id | UUID | 主键，关联 auth.users |
| email | TEXT | 邮箱（注册时同步） |
| username | TEXT | 用户名（唯一，URL 用） |
| nickname | TEXT | 显示昵称 |
| avatar_url | TEXT | 头像 URL（Storage） |
| bio | TEXT | 个人简介 |
| role | TEXT | 角色：member/verified_aviator/admin |
| created_at | TIMESTAMPTZ | 注册时间 |
| updated_at | TIMESTAMPTZ | 更新时间 |

### 2.2 flights · 飞行日志

```sql
CREATE TABLE flights (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  
  -- 基本信息
  date            DATE NOT NULL,
  flight_no       TEXT NOT NULL,                    -- 如 ZH9507
  airline         TEXT NOT NULL DEFAULT '',         -- 如 深圳航空
  airline_icao    TEXT NOT NULL DEFAULT '',         -- 如 CSZ
  aircraft        TEXT NOT NULL DEFAULT '',         -- 如 A350-900
  registration    TEXT NOT NULL DEFAULT '',         -- 如 B-304J
  seat            TEXT NOT NULL DEFAULT '',         -- 如 40L
  cabin_class     TEXT NOT NULL DEFAULT '',         -- 头等/公务/超经/经济
  
  -- 航段信息
  dep_icao        TEXT NOT NULL,                    -- 如 ZGSZ
  dep_city        TEXT NOT NULL DEFAULT '',
  arr_icao        TEXT NOT NULL,                    -- 如 ZSSS
  arr_city        TEXT NOT NULL DEFAULT '',
  via             TEXT NOT NULL DEFAULT '',         -- 经停
  
  -- 时间
  dep_time_scheduled  TEXT NOT NULL DEFAULT '',     -- 计划起飞 17:05
  dep_time_actual     TEXT NOT NULL DEFAULT '',     -- 实际起飞
  arr_time_scheduled  TEXT NOT NULL DEFAULT '',     -- 计划到达
  arr_time_actual     TEXT NOT NULL DEFAULT '',     -- 实际到达
  
  -- 专业字段（复刻纸质版）
  v1_vr_v2        TEXT NOT NULL DEFAULT '',         -- 如 145/150/155
  cruise_alt      TEXT NOT NULL DEFAULT '',         -- 巡航高度 FL360
  cruise_mach     TEXT NOT NULL DEFAULT '',         -- 巡航马赫数
  cruise_cas      TEXT NOT NULL DEFAULT '',         -- 巡航校准空速
  route           TEXT NOT NULL DEFAULT '',         -- 飞行航路
  sid             TEXT NOT NULL DEFAULT '',         -- 离场程序
  star            TEXT NOT NULL DEFAULT '',         -- 进场程序
  approach        TEXT NOT NULL DEFAULT '',         -- 进近方式
  weather_dep     TEXT NOT NULL DEFAULT '',         -- 出发天气
  weather_arr     TEXT NOT NULL DEFAULT '',         -- 到达天气
  
  -- 留言（复刻纸质版四级留言）
  remarks_passenger  TEXT NOT NULL DEFAULT '',      -- 乘客留言
  remarks_captain    TEXT NOT NULL DEFAULT '',      -- 机长留言
  remarks_fo         TEXT NOT NULL DEFAULT '',      -- 副机长留言
  remarks_purser     TEXT NOT NULL DEFAULT '',      -- 乘务长留言
  
  -- 元数据
  is_public       BOOLEAN NOT NULL DEFAULT false,
  distance_km     INTEGER,                          -- 自动计算
  flight_duration INTEGER,                          -- 分钟，自动计算
  created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at      TIMESTAMPTZ NOT NULL DEFAULT now()
);

COMMENT ON TABLE flights IS '飞行日志主表，复刻纸质版完整字段';
```

### 2.3 flight_tickets · 机票/票据图片

```sql
CREATE TABLE flight_tickets (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  flight_id   UUID NOT NULL REFERENCES flights(id) ON DELETE CASCADE,
  user_id     UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  image_url   TEXT NOT NULL,
  caption     TEXT NOT NULL DEFAULT '',
  sort_order  INTEGER NOT NULL DEFAULT 0,
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);

COMMENT ON TABLE flight_tickets IS '机票/票据图片，关联到飞行日志';
```

### 2.4 posts · 动态广场帖子（P1）

```sql
CREATE TABLE posts (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  title           TEXT NOT NULL DEFAULT '',
  content         TEXT NOT NULL,
  images          TEXT[] NOT NULL DEFAULT '{}',
  tags            TEXT[] NOT NULL DEFAULT '{}',
  flight_id       UUID REFERENCES flights(id),      -- 关联飞行日志（可选）
  likes_count     INTEGER NOT NULL DEFAULT 0,
  comments_count  INTEGER NOT NULL DEFAULT 0,
  views_count     INTEGER NOT NULL DEFAULT 0,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at      TIMESTAMPTZ NOT NULL DEFAULT now()
);

COMMENT ON TABLE posts IS '动态广场帖子，P1 阶段使用';
```

### 2.5 comments · 评论（P1）

```sql
CREATE TABLE comments (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  post_id     UUID NOT NULL REFERENCES posts(id) ON DELETE CASCADE,
  user_id     UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  content     TEXT NOT NULL,
  parent_id   UUID REFERENCES comments(id) ON DELETE CASCADE,  -- 楼中楼
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);

COMMENT ON TABLE comments IS '帖子评论，支持楼中楼回复';
```

### 2.6 follows · 关注关系（P2）

```sql
CREATE TABLE follows (
  follower_id   UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  followed_id   UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  PRIMARY KEY (follower_id, followed_id),
  CHECK (follower_id != followed_id)
);

COMMENT ON TABLE follows IS '用户关注关系';
```

### 2.7 likes · 点赞（P1）

```sql
CREATE TABLE likes (
  user_id      UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  target_type  TEXT NOT NULL CHECK (target_type IN ('post', 'flight', 'comment')),
  target_id    UUID NOT NULL,
  created_at   TIMESTAMPTZ NOT NULL DEFAULT now(),
  PRIMARY KEY (user_id, target_type, target_id)
);

COMMENT ON TABLE likes IS '点赞记录，支持帖子/飞行日志/评论';
```

## 4. RLS 行级权限策略

### 4.1 users 表

```sql
ALTER TABLE users ENABLE ROW LEVEL SECURITY;

-- 所有登录用户可查看其他用户的基本信息
CREATE POLICY "users_select_authenticated" ON users
  FOR SELECT TO authenticated USING (true);

-- 用户只能修改自己的资料
CREATE POLICY "users_update_own" ON users
  FOR UPDATE TO authenticated USING (auth.uid() = id) WITH CHECK (auth.uid() = id);

-- 用户只能插入自己的记录（注册时触发）
CREATE POLICY "users_insert_own" ON users
  FOR INSERT TO authenticated WITH CHECK (auth.uid() = id);
```

### 4.2 flights 表

```sql
ALTER TABLE flights ENABLE ROW LEVEL SECURITY;

-- 查看：公开的任何人可看，私有的只有所有者可看
CREATE POLICY "flights_select" ON flights
  FOR SELECT TO authenticated 
  USING (is_public = true OR user_id = auth.uid());

-- 匿名用户只能看公开的
CREATE POLICY "flights_select_anon" ON flights
  FOR SELECT TO anon USING (is_public = true);

-- 插入/更新/删除：只有所有者
CREATE POLICY "flights_insert_own" ON flights
  FOR INSERT TO authenticated WITH CHECK (user_id = auth.uid());

CREATE POLICY "flights_update_own" ON flights
  FOR UPDATE TO authenticated USING (user_id = auth.uid()) WITH CHECK (user_id = auth.uid());

CREATE POLICY "flights_delete_own" ON flights
  FOR DELETE TO authenticated USING (user_id = auth.uid());
```

### 4.3 flight_tickets 表

```sql
ALTER TABLE flight_tickets ENABLE ROW LEVEL SECURITY;

-- 查看：关联的 flight 是公开的，或所有者
CREATE POLICY "tickets_select" ON flight_tickets
  FOR SELECT USING (
    EXISTS (
      SELECT 1 FROM flights 
      WHERE flights.id = flight_tickets.flight_id 
      AND (flights.is_public = true OR flights.user_id = auth.uid())
    )
  );

-- 写入：只有 flight 所有者
CREATE POLICY "tickets_insert_own" ON flight_tickets
  FOR INSERT TO authenticated WITH CHECK (
    EXISTS (
      SELECT 1 FROM flights 
      WHERE flights.id = flight_tickets.flight_id 
      AND flights.user_id = auth.uid()
    )
  );

CREATE POLICY "tickets_update_own" ON flight_tickets
  FOR UPDATE TO authenticated USING (
    EXISTS (SELECT 1 FROM flights WHERE flights.id = flight_tickets.flight_id AND flights.user_id = auth.uid())
  );

CREATE POLICY "tickets_delete_own" ON flight_tickets
  FOR DELETE TO authenticated USING (
    EXISTS (SELECT 1 FROM flights WHERE flights.id = flight_tickets.flight_id AND flights.user_id = auth.uid())
  );
```

## 5. 触发器

### 5.1 自动创建 users 记录

```sql
CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO users (id, email, username, nickname)
  VALUES (
    NEW.id,
    NEW.email,
    split_part(NEW.email, '@', 1) || '_' || substr(NEW.id::text, 1, 4),
    COALESCE(NEW.raw_user_meta_data->>'nickname', split_part(NEW.email, '@', 1))
  );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();
```

### 5.2 自动更新 updated_at

```sql
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER flights_updated_at BEFORE UPDATE ON flights
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

CREATE TRIGGER users_updated_at BEFORE UPDATE ON users
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

CREATE TRIGGER posts_updated_at BEFORE UPDATE ON posts
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();
```

### 5.3 自动计算距离和时长

```sql
CREATE OR REPLACE FUNCTION calculate_flight_metrics()
RETURNS TRIGGER AS $$
BEGIN
  -- 如果有出发和到达机场，计算距离
  IF NEW.dep_icao IS NOT NULL AND NEW.arr_icao IS NOT NULL THEN
    SELECT 
      6371 * acos(cos(radians(a1.latitude)) * cos(radians(a2.latitude)) 
      * cos(radians(a2.longitude) - radians(a1.longitude)) 
      + sin(radians(a1.latitude)) * sin(radians(a2.latitude)))
    INTO NEW.distance_km
    FROM airports a1, airports a2
    WHERE a1.icao_code = NEW.dep_icao AND a2.icao_code = NEW.arr_icao;
  END IF;
  
  -- 如果有实际起飞和到达时间，计算时长
  -- （时间格式为 HH:MM 字符串，需解析）
  -- 暂时留空，由应用层计算
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER flights_calculate_metrics BEFORE INSERT OR UPDATE ON flights
  FOR EACH ROW EXECUTE FUNCTION calculate_flight_metrics();
```

### 5.4 同步 posts 的 likes_count / comments_count

```sql
-- 点赞时更新计数
CREATE OR REPLACE FUNCTION update_likes_count()
RETURNS TRIGGER AS $$
BEGIN
  IF TG_OP = 'INSERT' THEN
    IF NEW.target_type = 'post' THEN
      UPDATE posts SET likes_count = likes_count + 1 WHERE id = NEW.target_id;
    END IF;
    RETURN NEW;
  ELSIF TG_OP = 'DELETE' THEN
    IF OLD.target_type = 'post' THEN
      UPDATE posts SET likes_count = GREATEST(0, likes_count - 1) WHERE id = OLD.target_id;
    END IF;
    RETURN OLD;
  END IF;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER on_like_change AFTER INSERT OR DELETE ON likes
  FOR EACH ROW EXECUTE FUNCTION update_likes_count();

-- 评论时更新计数
CREATE OR REPLACE FUNCTION update_comments_count()
RETURNS TRIGGER AS $$
BEGIN
  IF TG_OP = 'INSERT' THEN
    UPDATE posts SET comments_count = comments_count + 1 WHERE id = NEW.post_id;
    RETURN NEW;
  ELSIF TG_OP = 'DELETE' THEN
    UPDATE posts SET comments_count = GREATEST(0, comments_count - 1) WHERE id = OLD.post_id;
    RETURN OLD;
  END IF;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER on_comment_change AFTER INSERT OR DELETE ON comments
  FOR EACH ROW EXECUTE FUNCTION update_comments_count();
```

## 6. 视图

### 6.1 用户统计视图

```sql
CREATE VIEW user_stats AS
SELECT 
  u.id AS user_id,
  u.nickname,
  COUNT(DISTINCT f.id) AS total_flights,
  COUNT(DISTINCT f.dep_icao) AS unique_dep_airports,
  COUNT(DISTINCT f.arr_icao) AS unique_arr_airports,
  COALESCE(SUM(f.distance_km), 0) AS total_distance_km,
  COALESCE(SUM(f.flight_duration), 0) AS total_flight_minutes,
  COUNT(DISTINCT f.aircraft) AS unique_aircraft_types
FROM users u
LEFT JOIN flights f ON f.user_id = u.id AND f.is_public = true
GROUP BY u.id, u.nickname;
```

### 6.2 公开广场视图

```sql
CREATE VIEW public_feed AS
SELECT 
  f.id,
  f.user_id,
  u.nickname,
  u.avatar_url,
  u.username,
  f.date,
  f.flight_no,
  f.airline,
  f.aircraft,
  f.registration,
  f.dep_icao,
  f.dep_city,
  f.arr_icao,
  f.arr_city,
  f.seat,
  f.cabin_class,
  f.distance_km,
  f.flight_duration,
  f.created_at,
  COUNT(t.id) AS ticket_count
FROM flights f
JOIN users u ON u.id = f.user_id
LEFT JOIN flight_tickets t ON t.flight_id = f.id
WHERE f.is_public = true
GROUP BY f.id, u.id
ORDER BY f.created_at DESC;
```

## 8. 迁移文件管理

迁移文件按顺序编号，存放在 `migrations/` 目录：

```
migrations/
├── 001_init_tables.sql          # 创建所有表
├── 002_init_indexes.sql         # 创建索引
├── 003_init_rls.sql             # 启用 RLS + 策略
├── 004_init_triggers.sql        # 触发器
├── 005_init_views.sql           # 视图
├── 006_seed_airports.sql        # 机场数据
├── 007_seed_aircraft.sql        # 机型数据
└── 008_import_flight_log.sql    # flight-log 数据迁移脚本
```

每个迁移文件头部包含：
```sql
-- Migration: 001_init_tables
-- Description: 创建 Windborne 所有数据表
-- Author: 小扶
-- Date: 2026-07-19
-- Depends on: -
```

import { Database } from '@/types/database'

## 10. 数据备份策略

### 10.1 自动备份

- Supabase 免费档：每日自动备份，保留 7 天

### 10.2 手动备份

每周通过 `pg_dump` 导出 SQL 备份到本地 + GitHub（私有仓库）：

```bash
pg_dump "postgresql://xxx:xxx@db.xxx.supabase.co:5432/postgres" \
  --no-owner --no-acl \
  -f backups/windborne_$(date +%Y%m%d).sql
```

### 10.3 恢复测试

每月测试一次从备份恢复到本地 Postgres，验证备份可用性。
